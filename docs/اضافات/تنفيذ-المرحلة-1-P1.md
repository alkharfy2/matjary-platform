## 1. حاجب الطلبات الوهمية (Fake Order Blocker)

### 1.1 نظرة عامة

| البند | التفاصيل |
|-------|----------|
| **ماهو** | نظام تقييم ثقة تلقائي على مستوى **المنصة بالكامل** — لما عميل يرفض الاستلام في متجر A، **كل المتاجر** تشوف التحذير |
| **ليه مهم** | أكبر مشكلة لتجار COD في مصر. يدفع شحن ذهاب + رجوع بسبب طلبات وهمية |
| **الميزة التنافسية** | **Shared Blacklist** — ولا منصة عربية (Easy Orders, Salla, YouCan) عندها ده |
| **بدون حزم** | لا نحتاج أي حزمة إضافية |

### 1.2 كيف يعمل

```
عميل برقم 01012345678 يطلب في متجر "أحمد"
    │
    ├─ لو أول مرة → "عميل جديد" — يمر عادي
    │
    ├─ لو عنده سجل سابق على المنصة:
    │   ├─ trust_score >= 70  → "عميل موثوق ✅" — يمر عادي
    │   ├─ trust_score 40-69  → "عميل عادي" — يمر مع ملاحظة
    │   ├─ trust_score 20-39  → "عميل مشكوك فيه ⚠️" — يمر + تحذير للتاجر
    │   └─ trust_score < 20   → خيارين:
    │       ├─ fakeOrderAutoReject = true  → "رقمك محظور — تواصل مع التاجر"
    │       └─ fakeOrderAutoReject = false → يمر + تحذير شديد للتاجر
    │
    └─ بعد استلام/رفض الطلب:
        ├─ delivered  → trust_score يزيد (+5 لكل طلب ناجح، حد أقصى 100)
        └─ cancelled/refunded (بسبب رفض الاستلام) → trust_score ينقص (-15 لكل رفض)
```

### 1.3 حساب الـ Trust Score

```typescript
/**
 * معادلة Trust Score:
 * 
 * score = 50 + (completed_orders × 5) - (rejected_orders × 15)
 * score = Math.max(0, Math.min(100, score))
 * 
 * - عميل جديد: يبدأ بـ 50 (neutral)
 * - كل طلب تم تسليمه: +5
 * - كل طلب مرفوض: -15 (عقوبة أعلى = ردع أقوى)
 * - الحد الأدنى: 0
 * - الحد الأقصى: 100
 * 
 * أمثلة:
 *   - عميل جديد: 50
 *   - 3 طلبات ناجحة + 0 رفض: 50 + 15 = 65
 *   - 5 طلبات ناجحة + 1 رفض: 50 + 25 - 15 = 60
 *   - 1 طلب ناجح + 2 رفض: 50 + 5 - 30 = 25 ⚠️
 *   - 0 طلبات ناجحة + 3 رفض: 50 - 45 = 5 🚫
 */
function calculateTrustScore(completedOrders: number, rejectedOrders: number): number {
  const raw = 50 + (completedOrders * 5) - (rejectedOrders * 15)
  return Math.max(0, Math.min(100, raw))
}
```

### 1.7 التنفيذ

#### دوال Trust Score

**ملف**: `src/lib/customers/trust-score.ts`

```typescript
import { createHash } from 'crypto'
import { db } from '@/db'
import { platformCustomers } from '@/db/schema'
import { eq } from 'drizzle-orm'

/** هاش رقم الموبايل للبحث */
export function hashPhone(phone: string): string {
  const cleaned = phone.replace(/[^0-9+]/g, '')
  return createHash('sha256').update(cleaned).digest('hex')
}

/** حساب Trust Score */
export function calculateTrustScore(completed: number, rejected: number): number {
  const raw = 50 + (completed * 5) - (rejected * 15)
  return Math.max(0, Math.min(100, raw))
}

/** الحصول على بيانات العميل أو إنشاء سجل جديد */
export async function getOrCreatePlatformCustomer(phone: string) {
  const hash = hashPhone(phone)
  
  // بحث عن العميل
  const existing = await db
    .select()
    .from(platformCustomers)
    .where(eq(platformCustomers.phoneHash, hash))
    .limit(1)
  
  if (existing[0]) return existing[0]
  
  // إنشاء عميل جديد
  const [created] = await db
    .insert(platformCustomers)
    .values({
      phone: phone.replace(/[^0-9+]/g, ''),
      phoneHash: hash,
    })
    .onConflictDoNothing()
    .returning()
  
  // لو حصل race condition والسجل اتعمل من طلب تاني
  if (!created) {
    const [found] = await db
      .select()
      .from(platformCustomers)
      .where(eq(platformCustomers.phoneHash, hash))
      .limit(1)
    return found!
  }
  
  return created
}

/** تحديث Trust Score بعد تغيير حالة الطلب */
export async function updateCustomerTrustScore(
  phone: string, 
  event: 'delivered' | 'rejected',
) {
  const hash = hashPhone(phone)
  
  const [customer] = await db
    .select()
    .from(platformCustomers)
    .where(eq(platformCustomers.phoneHash, hash))
    .limit(1)
  
  if (!customer) return
  
  const newCompleted = customer.completedOrders + (event === 'delivered' ? 1 : 0)
  const newRejected = customer.rejectedOrders + (event === 'rejected' ? 1 : 0)
  const newScore = calculateTrustScore(newCompleted, newRejected)
  
  await db
    .update(platformCustomers)
    .set({
      totalOrders: customer.totalOrders, // لا يتغير هنا — يتغير عند الـ checkout
      completedOrders: newCompleted,
      rejectedOrders: newRejected,
      trustScore: newScore,
      // حظر تلقائي لو السكور وصل 0
      isBlocked: newScore === 0 ? true : customer.isBlocked,
      blockReason: newScore === 0 ? 'رفض متكرر — حظر تلقائي' : customer.blockReason,
    })
    .where(eq(platformCustomers.id, customer.id))
}

/** Trust level labels */
export function getTrustLevel(score: number): {
  label: string
  labelAr: string
  color: 'green' | 'yellow' | 'orange' | 'red'
  emoji: string
} {
  if (score >= 70) return { label: 'trusted', labelAr: 'عميل موثوق', color: 'green', emoji: '✅' }
  if (score >= 40) return { label: 'normal', labelAr: 'عميل عادي', color: 'yellow', emoji: '👤' }
  if (score >= 20) return { label: 'suspicious', labelAr: 'عميل مشكوك فيه', color: 'orange', emoji: '⚠️' }
  return { label: 'high_risk', labelAr: 'عميل عالي الخطورة', color: 'red', emoji: '🚫' }
}
```

#### فحص العميل عند Checkout

**ملف**: `src/lib/customers/customer-check.ts`

```typescript
import { db } from '@/db'
import { storeCustomerBlocks } from '@/db/schema'
import { and, eq } from 'drizzle-orm'
import { getOrCreatePlatformCustomer, getTrustLevel, hashPhone } from './trust-score'

type CustomerCheckResult = {
  allowed: boolean
  customer: {
    trustScore: number
    trustLevel: ReturnType<typeof getTrustLevel>
    totalOrders: number
    completedOrders: number
    rejectedOrders: number
    isBlocked: boolean
  }
  rejectReason?: string
}

/**
 * فحص العميل قبل إنشاء الطلب
 * 
 * يُستدعى من checkout/route.ts
 */
export async function checkCustomerTrust(
  phone: string,
  storeId: string,
  settings: {
    fakeOrderBlockerEnabled?: boolean
    fakeOrderMinTrustScore?: number
    fakeOrderAutoReject?: boolean
  },
): Promise<CustomerCheckResult> {
  // لو الميزة مش مفعّلة — يمر عادي
  if (settings.fakeOrderBlockerEnabled === false) {
    return {
      allowed: true,
      customer: {
        trustScore: 50,
        trustLevel: getTrustLevel(50),
        totalOrders: 0,
        completedOrders: 0,
        rejectedOrders: 0,
        isBlocked: false,
      },
    }
  }
  
  // 1. فحص الحظر على مستوى المنصة
  const platformCustomer = await getOrCreatePlatformCustomer(phone)
  
  if (platformCustomer.isBlocked) {
    return {
      allowed: false,
      customer: {
        trustScore: platformCustomer.trustScore,
        trustLevel: getTrustLevel(platformCustomer.trustScore),
        totalOrders: platformCustomer.totalOrders,
        completedOrders: platformCustomer.completedOrders,
        rejectedOrders: platformCustomer.rejectedOrders,
        isBlocked: true,
      },
      rejectReason: 'هذا الرقم محظور من الطلب على المنصة',
    }
  }
  
  // 2. فحص الحظر على مستوى المتجر
  const cleanPhone = phone.replace(/[^0-9+]/g, '')
  const [storeBlock] = await db
    .select()
    .from(storeCustomerBlocks)
    .where(and(
      eq(storeCustomerBlocks.storeId, storeId),
      eq(storeCustomerBlocks.customerPhone, cleanPhone),
    ))
    .limit(1)
  
  if (storeBlock) {
    return {
      allowed: false,
      customer: {
        trustScore: platformCustomer.trustScore,
        trustLevel: getTrustLevel(platformCustomer.trustScore),
        totalOrders: platformCustomer.totalOrders,
        completedOrders: platformCustomer.completedOrders,
        rejectedOrders: platformCustomer.rejectedOrders,
        isBlocked: true,
      },
      rejectReason: 'هذا الرقم محظور من الطلب في هذا المتجر',
    }
  }
  
  // 3. فحص Trust Score
  const minScore = settings.fakeOrderMinTrustScore ?? 20
  const autoReject = settings.fakeOrderAutoReject ?? false
  
  if (platformCustomer.trustScore < minScore && autoReject) {
    return {
      allowed: false,
      customer: {
        trustScore: platformCustomer.trustScore,
        trustLevel: getTrustLevel(platformCustomer.trustScore),
        totalOrders: platformCustomer.totalOrders,
        completedOrders: platformCustomer.completedOrders,
        rejectedOrders: platformCustomer.rejectedOrders,
        isBlocked: false,
      },
      rejectReason: 'لا يمكن إتمام الطلب — يرجى التواصل مع المتجر',
    }
  }
  
  // ✅ مسموح (ممكن يكون مع تحذير للتاجر)
  return {
    allowed: true,
    customer: {
      trustScore: platformCustomer.trustScore,
      trustLevel: getTrustLevel(platformCustomer.trustScore),
      totalOrders: platformCustomer.totalOrders,
      completedOrders: platformCustomer.completedOrders,
      rejectedOrders: platformCustomer.rejectedOrders,
      isBlocked: false,
    },
  }
}
```

## 2. استرجاع السلات المتروكة (Abandoned Cart Recovery)

### 2.1 نظرة عامة

| البند | التفاصيل |
|-------|----------|
| **ماهو** | لما عميل يبدأ checkout ويدخل بياناته (اسم + رقم) بس ميكملش الطلب — نحفظ السلة ونبعتله رسالة تذكير |
| **ليه مهم** | 70% من السلات بتتسيب. استرجاع 10% منهم = زيادة مبيعات ملموسة |
| **التكلفة** | مجاني — WhatsApp عبر wa.me link (التاجر يبعت يدوي أو تلقائي) |
| **بدون حزم** | لا نحتاج أي حزمة — API + database + cron-like logic |

### 2.2 كيف يعمل

```
عميل يفتح Checkout → يملأ اسمه ورقمه → يختار محافظة
    │
    ├── يكمل الطلب → ✅ عادي (مش سلة متروكة)
    │
    └── يسيب الصفحة بدون إكمال → بعد 5 دقائق من عدم النشاط:
        │
        ├── السلة تتحفظ في store_abandoned_carts
        │   (الاسم، الرقم، الإيميل، المنتجات، المبلغ)
```

#### API لحفظ السلة المتروكة

**ملف**: `src/app/api/storefront/abandoned-cart/route.ts`

```typescript
import { NextRequest } from 'next/server'
import { db } from '@/db'
import { storeAbandonedCarts, storeOrders } from '@/db/schema'
import { getCurrentStore } from '@/lib/tenant/get-current-store'
import { and, eq, gt } from 'drizzle-orm'

export async function POST(request: NextRequest) {
  try {
    const store = await getCurrentStore()
    if (!store) return new Response(null, { status: 404 })
    
    // لو الميزة مش مفعّلة
    if (store.settings?.abandonedCartEnabled === false) {
      return new Response(null, { status: 204 })
    }
    
    const body = await request.json()
    const { customerName, customerPhone, customerEmail, items } = body
    
    if (!customerName || !customerPhone || !items?.length) {
      return new Response(null, { status: 400 })
    }
    
    // التأكد إنه مفيش طلب اتعمل بالفعل بنفس الرقم في آخر 10 دقائق
    const tenMinutesAgo = new Date(Date.now() - 10 * 60 * 1000)
    const [recentOrder] = await db
      .select({ id: storeOrders.id })
      .from(storeOrders)
      .where(and(
        eq(storeOrders.storeId, store.id),
        eq(storeOrders.customerPhone, customerPhone),
        gt(storeOrders.createdAt, tenMinutesAgo),
      ))
      .limit(1)
    
    if (recentOrder) {
      // العميل كمّل طلبه — مش abandoned
      return new Response(null, { status: 204 })
    }
    
    // التأكد إنه مفيش سلة متروكة بنفس الرقم في آخر ساعة (منع التكرار)
    const oneHourAgo = new Date(Date.now() - 60 * 60 * 1000)
    const [existingCart] = await db
      .select({ id: storeAbandonedCarts.id })
      .from(storeAbandonedCarts)
      .where(and(
        eq(storeAbandonedCarts.storeId, store.id),
        eq(storeAbandonedCarts.customerPhone, customerPhone),
        gt(storeAbandonedCarts.createdAt, oneHourAgo),
      ))
      .limit(1)
    
    if (existingCart) {
      // حدّث السلة الموجودة بدل ما نعمل واحدة جديدة
      await db
        .update(storeAbandonedCarts)
        .set({ items, subtotal: items.reduce((sum: number, i: { unitPrice: number; quantity: number }) => sum + i.unitPrice * i.quantity, 0).toFixed(2) })
        .where(eq(storeAbandonedCarts.id, existingCart.id))
      return new Response(null, { status: 204 })
    }
    
    const subtotal = items.reduce(
      (sum: number, i: { unitPrice: number; quantity: number }) => sum + i.unitPrice * i.quantity, 0
    )
    
    await db.insert(storeAbandonedCarts).values({
      storeId: store.id,
      customerName,
      customerPhone,
      customerEmail: customerEmail || null,
      items,
      subtotal: subtotal.toFixed(2),
      expiresAt: new Date(Date.now() + 7 * 24 * 60 * 60 * 1000), // 7 أيام
    })
    
    return new Response(null, { status: 201 })
  } catch {
    return new Response(null, { status: 500 })
  }
}
```

```typescript
type UpsellOfferProps = {
  orderId: string
  orderNumber: string
  offer: {
    ruleId: string
    product: {
      id: string
      name: string
      slug: string
      price: number
      compareAtPrice: number | null
      images: string[]
    }
    discountType: 'percentage' | 'fixed'

export function UpsellOffer({ orderId, orderNumber, offer }: UpsellOfferProps) {
```

```typescript
export async function getCrossSellProducts(
  storeId: string,
  productId: string,
  options?: { limit?: number },
) {
  const limit = options?.limit ?? 4
  
  const relations = await db
    .select({
      relatedProduct: {
        id: storeProducts.id,
        name: storeProducts.name,
        slug: storeProducts.slug,
        price: storeProducts.price,
        compareAtPrice: storeProducts.compareAtPrice,
        images: storeProducts.images,
      },
    })
    .from(storeProductRelations)
    .innerJoin(storeProducts, eq(storeProductRelations.relatedProductId, storeProducts.id))
    .where(and(
      eq(storeProductRelations.storeId, storeId),
      eq(storeProductRelations.productId, productId),
      eq(storeProductRelations.relationType, 'cross_sell'),
      eq(storeProducts.isActive, true),
    ))
    .orderBy(storeProductRelations.sortOrder)
    .limit(limit)
  
  return relations.map(r => r.relatedProduct)
}
```

```typescript
'use client'

import { useEffect, useState, useCallback } from 'react'
import { useStore } from '@/lib/tenant/store-context'
import { usePathname } from 'next/navigation'

export function ExitIntentPopup() {
  const store = useStore()
  const pathname = usePathname()
  const [show, setShow] = useState(false)
  const [copied, setCopied] = useState(false)
  
  const settings = store.settings
  
  // تحديد هل الصفحة الحالية ضمن الصفحات المستهدفة
  const isTargetPage = useCallback(() => {
    const pages = settings.exitIntentPages ?? []
    if (pages.includes('product') && pathname.startsWith('/product/')) return true
    if (pages.includes('cart') && pathname === '/cart') return true
    if (pages.includes('checkout') && pathname === '/checkout') return true
    return false
  }, [pathname, settings.exitIntentPages])
  
  useEffect(() => {
    if (!settings.exitIntentEnabled) return
    if (!settings.exitIntentCouponCode) return
    if (!isTargetPage()) return
    
    // تحقق إنه ما ظهرش قبل كده في الجلسة
    const shown = sessionStorage.getItem('exit-intent-shown')
    if (shown) return
    
    // Desktop: mouseout من أعلى الصفحة
    const handleMouseOut = (e: MouseEvent) => {
      if (e.clientY <= 0) {
        sessionStorage.setItem('exit-intent-shown', '1')
        setShow(true)
      }
    }
    
    // تأخير 5 ثواني قبل تفعيل الحدث (منع ظهور فوري)
    const timer = setTimeout(() => {
      document.addEventListener('mouseout', handleMouseOut)
    }, 5000)
    
    return () => {
      clearTimeout(timer)
      document.removeEventListener('mouseout', handleMouseOut)
    }
  }, [settings, isTargetPage])
  
  const handleCopy = async () => {
    if (settings.exitIntentCouponCode) {
      await navigator.clipboard.writeText(settings.exitIntentCouponCode)
      setCopied(true)
      setTimeout(() => setCopied(false), 2000)
    }
  }
  
  if (!show) return null
  
  const message = settings.exitIntentMessage ?? 'مستنيك! خصم خاص على طلبك لو كمّلت دلوقتي'
  
  return (
    <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/50 p-4">
      <div className="w-full max-w-sm rounded-2xl bg-white p-8 text-center shadow-2xl">
        <p className="mb-2 text-4xl">🎉</p>
        <h3 className="mb-3 text-xl font-bold">{message}</h3>
        
        {settings.exitIntentCouponCode && (
          <div className="mb-4">
            <p className="mb-2 text-sm text-gray-500">كود الخصم:</p>
            <button
              onClick={handleCopy}
              className="rounded-lg border-2 border-dashed border-gray-300 bg-gray-50 px-6 py-3 font-mono text-lg font-bold tracking-wider transition-colors hover:border-gray-400"
            >
              {settings.exitIntentCouponCode}
              <span className="ms-2 text-sm">{copied ? '✅' : '📋'}</span>
            </button>
          </div>
        )}
        
        <div className="flex flex-col gap-2">
          <a
            href="/checkout"
            className="rounded-lg bg-[var(--store-primary)] px-6 py-3 font-medium text-white"
          >
            كمّل الطلب →
          </a>
          <button
            onClick={() => setShow(false)}
            className="text-sm text-gray-400 hover:text-gray-600"
          >
            لا شكراً
          </button>
        </div>
      </div>
    </div>
  )
}
```

```typescript
  if (typeof window !== 'undefined' && typeof window.snaptr === 'function') {
```
